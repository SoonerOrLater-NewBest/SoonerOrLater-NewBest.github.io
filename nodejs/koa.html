# 上手 Koa2 + MySQL 开发

  • 上手 Koa2 + MySQL 开发

  • 罗列使用的 koa2 中间件和配置注释

const Koa = require('koa');
const router = require('koa-router')();
const bodyParser = require('koa-bodyparser');
const apiRouter = require('./router');
const path = require('path');
const views = require('koa-views');
const cors = require('koa2-cors');

const https = require('https');
const fs = require('fs');
// const sslify = require('koa-sslify').default;

const app = new Koa();

// Force HTTPS on all page
// app.use(sslify());
// SSL options
let options = {
  key: fs.readFileSync('./ssl/server.key'), // ssl文件路径
  cert: fs.readFileSync('./ssl/server.pem') // ssl文件路径
};

app.use(
  cors({
    origin: function (ctx) {
      if (ctx.url === '/api/system/settings') {
        return '*'; // 允许来自所有域名请求
      }
      return 'www.twoto.top';
    },
    exposeHeaders: ['WWW-Authenticate', 'Server-Authorization'],
    maxAge: 5, // 指定本次预检请求的有效期
    credentials: true, // 是否允许发送Cookie
    allowMethods: ['GET', 'POST', 'DELETE'], // 设置允许的HTTP请求类型
    allowHeaders: ['Content-Type', 'Authorization', 'Accept'] //支持的所有头信息字段
  })
);
// 引入地址变量和静态资源中间件
const staticPath = require('koa-static');
// 使用访问静态中间件，staticPath 是地址 文件夹
app.use(staticPath(path.join(__dirname, './static')));

// 加载模板引擎
// './view' 是文件夹的路径，也就是模板所在的位置
app.use(
  views(path.join(__dirname, './views'), {
    extension: 'ejs'
  })
);
// 调用模板引擎
// 首页
const index = router
  .get('/', async (ctx) => {
    let title = '学习使我快乐';
    // 通过ctx.render调用的方法模板，index是模板的名称，title是传递的东西
    await ctx.render('index', {
      title
    });
  })
  .routes();

app.use(index);
app.use(bodyParser());
app.use(apiRouter.routes());
app.use(apiRouter.allowedMethods());
// app.listen(80);
// console.log('监听80');
https.createServer(options, app.callback()).listen(443);
console.log('node-https服务已启动');
